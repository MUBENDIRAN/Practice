reverse integer
class Solution(object):
    def reverse(self, x):
        n=abs(x)
        rev=0
        while n > 0:
            N = n % 10
            rev = rev*10 + N
            n = n // 10

        if x < 0:
            rev = -rev

        if rev < -2**31 or rev > 2**31 -1:
            return 0

        return rev

reverse integer
class Solution(object):
    def reverse(self, x):
        n=abs(x)
        rev=0
        while n > 0:
            N = n % 10
            rev = rev*10 + N
            n = n // 10

        if x < 0:
            rev = -rev

        if rev < -2**31 or rev > 2**31 -1:
            return 0

        return rev

palindrome
class Solution:
    def isPalindrome(self, x: int) -> bool:
        n = x
        org=n
        rev=0

        while n > 0:
            N = n % 10
            rev = rev*10 + N
            n = n // 10

        if org == rev:
            return True
        else:
            return False

Richest customer 
class Solution:
    def maximumWealth(self, accounts: List[List[int]]) -> int:
        rich = 0
        for i in accounts:
            yes = sum(i)
            if yes > rich:
                rich = yes
        return rich  

reverse list of character
class Solution:
    def reverseString(self, s: List[str]) -> None:
        left = 0
        right = len(s) -1 
        while left < right:
            s[left],s[right]=s[right],s[left]
            left+=1
            right-=1

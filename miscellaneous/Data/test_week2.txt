TWO SUM
class Solution:
    def twoSum(self, numbers: List[int], target: int) -> List[int]:
        left = 0
        right = len(numbers) - 1
        two_sum = 0
        while left <= right:
            two_sum = numbers[left] + numbers[right]
            if two_sum == target:
                return left+1,right+1
                break
            elif two_sum < target:
                left+=1
            else:
                right-=1
            
MOVE ZEROS
class Solution:
    def moveZeroes(self, nums: List[int]) -> None:
        pos = 0
        for i in range(0,len(nums)):
            if nums[i] != 0:
                nums[pos] = nums[i]
                pos+=1
        for i in range(pos,len(nums)):
            nums[i] = 0
REMOVE DUPLICATE FROM SORTED ARRAY
class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        pos = 1
        for i in range(1,len(nums)):
            if nums[i] != nums[i-1]:
                nums[pos] = nums[i]
                pos+=1
        return pos
SQUARES OF SORTED ARRAY
class Solution:
    def sortedSquares(self, nums: List[int]) -> List[int]:
        result = [0]*len(nums)
        left= 0
        right = len(nums) - 1
        pos = len(nums) - 1
        while left <= right:
            if abs(nums[left]) > abs(nums[right]):
                result[pos] = nums[left]*nums[left]
                left+=1
            else:
                result[pos] = nums[right]*nums[right]
                right-=1
            pos-=1
        return result 
FIND THE ANAGRAM
class Solution:
    def findAnagrams(self, s: str, p: str) -> List[int]:
        res = []
        freq_p ={}
        for ch in p:
            freq_p[ch] = freq_p.get(ch,0) + 1
        freq_s ={}
        start = 0
        for end in range(len(s)):
            freq_s[s[end]] = freq_s.get(s[end],0) + 1
            if end - start + 1 == len(p):
                if freq_p == freq_s:
                    res.append(start)
                ch = s[start]
                if ch in freq_s:
                    freq_s[ch]-=1
                    if freq_s[ch] == 0:
                        del freq_s[ch]
                start+=1
        return res
LENGTH OF LONGEST SUBSTRING
class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        last_seen = {}
        start = 0
        max_len = 0
        for end in range(len(s)):
            if s[end] in last_seen and last_seen[s[end]] >= start:
                start = last_seen[s[end]] + 1
            last_seen[s[end]] = end
            max_len = max(max_len,end - start + 1)
        return max_len        
FIRST AND LAST POSITION 
class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        first = -1
        left = 0
        right = len(nums) - 1
        while left <= right:
            mid = (left + right) // 2
            if nums[mid] == target:
                first = mid
                right = mid - 1
            elif nums[mid] < target:
                left = mid + 1
            else:
                right = mid - 1
        left = 0
        right = len(nums) - 1
        last = -1
        while left <= right:
            mid = (left + right) // 2
            if nums[mid] == target:
                last = mid
                left = mid + 1
            elif nums[mid] < target:
                left = mid + 1
            else:
                right = mid - 1
        return first,last
FIRST NEGATIVES INTEGER OF K
#User function Template for python3

class Solution:
    def firstNegInt(self, arr, k): 
        negatives = []
        start = 0
        ans = []
        for end in range(0,len(arr)):
            if arr[end] < 0:
                negatives.append(end)
            if end - start + 1 == k:
                if negatives:
                    ans.append(arr[negatives[0]])
                else:
                    ans.append(0)
                if negatives and negatives[0] == start:
                    negatives.pop(0)
                start+=1
        return ans
MOUNTAIN PEAK
class Solution:
    def peakIndexInMountainArray(self, arr: List[int]) -> int:
        left = 0
        right = len(arr) - 1
        while left < right:
            mid = (left + right) // 2
            if arr[mid] < arr[mid + 1]:
                left = mid + 1
            else:
                right = mid 
        return left
